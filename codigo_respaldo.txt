// ELIMINANDO PRODUCTOS REPETIDOS

    const cart = [...new Set(shoppingCart)]

    // LIMPIANDO INTERFAZ

    function cleanModal() {
        cartContainer.innerHTML = ''
    }

    // MOSTRANDO ITEMS EN LA INTERFAZ

    function shoppingRender() {

        cleanModal()

        let spanCart = document.querySelector('#spanCart').innerText = cart.length

        cart.forEach((producto) => {

            let itemContainer = document.createElement('div')
            itemContainer.classList.add('cart-product-container', 'd-flex', 'justify-content-between', 'align-items-center', 'py-3', 'my-3')
                    
            itemContainer.innerHTML = `
            
                        <div class="d-flex justify-content-start align-items-center col-8 m-0">
                            <div class="cart-product-img d-flex justify-content-center align-items-center mx-3">
                                <img class="img-fluid" src="${producto.imagen}" alt="">
                            </div>
                            <div class="cart-product-name d-flex justify-content-center align-items-center">
                                <p class="m-0">${producto.nombre}</p>
                            </div>
                        </div>
    
                        <div class="d-flex justify-content-center align-items-center">
                            <div class="cart-product-quantity d-flex justify-content-center align-items-center">
                                <i class="minus-button fa-solid fa-minus"></i>
                                <p class="p-3 m-0 quantity">1</p>
                                <i class="plus-button fa-solid fa-plus"></i>
                            </div>
                            <div class="p-2 text-center">
                                <i class="text-end cart-close-product fa-solid fa-xmark"></i>
                            </div>
                        </div>
                    `
                cartContainer.appendChild(itemContainer)
        })

        // FUNCIONALIDAD DE LOS BOTONES

        let quantity = document.querySelector('.quantity')
        let botonMas = document.querySelectorAll('.plus-button')
        botonMas.forEach((e) => {
           e.onclick = () => {
            quantity.innerText++
            

           }
        })

        let botonMenos = document.querySelectorAll('.minus-button').forEach((e) => {
            e.onclick = () =>{
                quantity.innerText--
            }
        })

    }

    // MOSTRANDO EL PRECIO TOTAL

    function precioTotal() {

        let totalCompra = cart.reduce((acc, prod) => acc + prod.precio, 0)

        let cartPriceTotal = document.getElementById('cartPriceTotal').innerText = 'Total a pagar:  ' + '   ' + totalCompra + '$'
    }

    // BOTONES DEL MODAL

    navCart.onclick = (e) => {
        e.preventDefault()
        shoppingModal.style.transform = 'translateX(0px)'
    }

    shoppingCloseButton.onclick = () => {
        shoppingModal.style.transform = 'translateX(3000px)'
    }
    let vaciarCarrito = document.querySelector('#borrarTodo').onclick = () => {
        
        cartContainer.innerHTML = `
                <!-- texto "vacío" -->
                <div class="cart-item-container">
                    <h3 class="text-center empty-text">vacío</h3>
                </div>

                <!-- Precio total -->      
        `
        cart.splice(0, cart.length)
        shoppingCart.splice(0, shoppingCart.length)
        spanCart.innerText = cart.length
        precioTotal()
    }

    shoppingRender()
    precioTotal()

    console.log(itemContainer)

}
